			function showHide(thisDiv,update)
			{
				if (thisDiv != null)
				{
					var icon = thisDiv.getElementsByTagName('img')[0];
					var id = thisDiv.id + 'Details';

						icon.src = 'images/opened.gif';

					}

						icon.src = 'images/closed.gif';

					}

						updateHidden();
